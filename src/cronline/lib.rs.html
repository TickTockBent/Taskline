1//! # Cronline: Lightweight Task Scheduling for Rust
2//!
3//! Cronline is a minimalist yet powerful cron-style task scheduler built with Rust.
4//! Designed for simplicity, performance, and ease of integration, it enables developers
5//! to effortlessly manage scheduled asynchronous tasks within Rust applications.
6//!
7//! ## Features
8//!
9//! - **Cron-like Scheduling**: Intuitive scheduling using standard cron syntax
10//! - **Interval-based Scheduling**: Simple interval-based task execution
11//! - **Async-Await Native**: First-class async task support leveraging Tokio
12//! - **Robust Error Handling**: Configurable retries and timeout handling
13//! - **Task Statistics**: Built-in execution tracking and metrics
14//! - **Pause/Resume**: Dynamic task control without stopping the scheduler
15//! - **Task Tags/Labels**: Organize and filter tasks using tags
16//! - **Graceful Cancellation**: Cancel running tasks cleanly
17//! - **Timeout Warnings**: Get warned at 80% of timeout duration
18//! - **Event Bus**: Subscribe to scheduler and task lifecycle events
19//! - **Auto-generated Names**: Meaningful task names from cron expressions
20//! - **Thread-Safe**: All components are safe to share across threads
21//!
22//! ## Quick Start
23//!
24//! Add Cronline to your `Cargo.toml`:
25//!
26//! ```toml
27//! [dependencies]
28//! cronline = "0.2.1"
29//! tokio = { version = "1", features = ["full"] }
30//! ```
31//!
32//! ## Basic Example
33//!
34//! ```no_run
35//! use cronline::{Scheduler, Task};
36//!
37//! #[tokio::main]
38//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
39//!     // Create a new scheduler
40//!     let scheduler = Scheduler::new();
41//!
42//!     // Add a task that runs every hour at minute 0
43//!     scheduler.add("0 * * * *", Task::new(|| async {
44//!         println!("Running hourly task!");
45//!         Ok(())
46//!     }).with_name("Hourly Task")).await?;
47//!
48//!     // Start the scheduler
49//!     scheduler.start().await?;
50//!
51//!     // Let it run for a while
52//!     tokio::time::sleep(tokio::time::Duration::from_secs(3600)).await;
53//!
54//!     // Stop the scheduler gracefully
55//!     scheduler.stop().await?;
56//!
57//!     Ok(())
58//! }
59//! ```
60//!
61//! ## Advanced Usage
62//!
63//! ```no_run
64//! use cronline::{Scheduler, Task, TaskConfig, SchedulerConfig};
65//! use std::time::Duration;
66//!
67//! #[tokio::main]
68//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
69//!     // Create scheduler with custom configuration
70//!     let scheduler = Scheduler::with_config(SchedulerConfig {
71//!         check_interval: Duration::from_millis(100),
72//!         continue_on_error: true,
73//!         shutdown_grace_period: Duration::from_secs(10),
74//!     });
75//!
76//!     // Subscribe to events
77//!     let mut events = scheduler.event_bus().subscribe();
78//!     tokio::spawn(async move {
79//!         while let Ok(event) = events.recv().await {
80//!             println!("Event: {:?}", event);
81//!         }
82//!     });
83//!
84//!     // Add task with custom configuration
85//!     scheduler.add("*/5 * * * *", Task::new(|| async {
86//!         // Your task logic here
87//!         Ok(())
88//!     })
89//!     .with_name("Custom Task")
90//!     .with_tags(&["backup", "critical"])
91//!     .with_config(TaskConfig {
92//!         timeout: Some(Duration::from_secs(30)),
93//!         max_retries: 3,
94//!         retry_delay: Duration::from_secs(5),
95//!         fail_scheduler_on_error: false,
96//!     })).await?;
97//!
98//!     // Add interval-based task
99//!     scheduler.add_task(Task::new(|| async {
100//!         println!("Running every 5 minutes!");
101//!         Ok(())
102//!     })
103//!     .with_interval(Duration::from_secs(300))
104//!     .with_tag("monitoring")).await?;
105//!
106//!     scheduler.start().await?;
107//!
108//!     // Filter tasks by tag
109//!     let critical_tasks = scheduler.tasks_with_tag("critical").await;
110//!     println!("Found {} critical tasks", critical_tasks.len());
111//!
112//!     // Do other work...
113//!
114//!     scheduler.stop().await?;
115//!     Ok(())
116//! }
117//! ```
118//!
119//! ## Cron Expression Syntax
120//!
121//! Cronline uses standard cron syntax: `* * * * *` (minute, hour, day of month, month, day of week)
122//!
123//! Examples:
124//! - `* * * * *` - Every minute
125//! - `0 * * * *` - Every hour at minute 0
126//! - `0 0 * * *` - Every day at midnight
127//! - `0 12 * * MON-FRI` - Weekdays at noon
128//! - `*/15 * * * *` - Every 15 minutes
129//!
